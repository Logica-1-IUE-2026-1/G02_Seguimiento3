INICIO FUNCION celsius_a_fahrenheit(c)
 DEFINIR
 resultado = c * 9 / 5 + 32
 Mostrar resultado
FIN FUNCION

INICIO FUNCION fahrenheit_a_celsius(f)
 DEFINIR
 resultado = (f - 32) * 5 / 9
 Mostrar resultado
FIN FUNCION
